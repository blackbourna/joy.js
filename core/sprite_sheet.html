Source code

var engine = new Joy.Engine({ canvas2d: document.getElementById('example') });
var rufus;
engine.createScene(function(scene) {

  rufus = new Joy.SpriteSheet({
    x: engine.width / 2,
    y: engine.height / 2,
    src: "/assets/examples/rufus/spritesheets/rufus.png",

    // Size of a single frame
    width: 120,
    height: 120,

    // Animation frames
    animations: {
      "walking": [0, 30],
      "idle": [31, 39],
      "jumping": [40, 59],
      "falling": [60, 67],
    }
  }).bind('load', function() {
    // Centralize position when loaded
    this.position.x -= this.width / 2;
    this.position.y -= this.height / 2;
  });

  rufus.play("walking");
  scene.addChild(rufus);
});
