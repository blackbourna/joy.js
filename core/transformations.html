var engine = new Joy.Engine({ canvas2d: document.getElementById('example') });
engine.createScene(function(scene) {
  var logo = new Joy.Sprite("/assets/examples/joyjs.png").bind('load', function () {
    this.pivot.x = this.width / 2;
    this.pivot.y = this.height / 2;
    this.position.x = engine.width / 2;
    this.position.y = engine.height - this.height;
    this.scale.x = 0.0;
    this.scale.y = 0.0;
    this.rotation = -180;
    this.alpha = 0;

    // Use tweens to manipulate DisplayObject transformations
    new Joy.Tween(this.position).to({y: engine.height / 2 + 50}, 3000).easing( Joy.TweenManager.Easing.Elastic.Out ).start();
    new Joy.Tween(this.scale).to({x: 1.0, y: 1.0}, 2000).easing( Joy.TweenManager.Easing.Exponential.Out ).start();
    new Joy.Tween(this).to({rotation: 0.0, alpha: 1}, 3000).easing( Joy.TweenManager.Easing.Elastic.Out ).start();
  });

  var joy = new Joy.Sprite("/assets/examples/joyjs-text.png").bind('load', function () {
    this.pivot.x = this.width / 2;
    this.pivot.y = this.height / 2;
    this.position.x = engine.width / 2;
    this.position.y = engine.height / 2;
    this.scale.x = 0.5;
    this.scale.y = 0.5;
    this.alpha = 0;

    // Use tweens to manipulate DisplayObject transformations
    new Joy.Tween(this.position).to({y: this.position.y - 150}, 5000).easing( Joy.TweenManager.Easing.Elastic.Out ).start();
    new Joy.Tween(this.scale).to({x: 0.9, y: 0.9}, 2000).easing( Joy.TweenManager.Easing.Exponential.Out ).start();
    new Joy.Tween(this).to({alpha: 1}, 3000).easing( Joy.TweenManager.Easing.Exponential.Out ).start();
  });

  scene.addChild(logo);
  scene.addChild(joy);
});
